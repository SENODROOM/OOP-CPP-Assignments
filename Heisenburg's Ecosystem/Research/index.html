// ShoppingCartManagementSystem.h — Linked List + OOP Design

struct Attribute {
    char* name;   // e.g. "color", "size"
    char* value;  // e.g. "red", "XL"
    Attribute* next;

    Attribute(const char* n, const char* v) {
        name  = new char[strlen(n) + 1]; strcpy(name, n);
        value = new char[strlen(v) + 1]; strcpy(value, v);
        next = nullptr;
    }
    ~Attribute() { delete[] name; delete[] value; }
};

struct CartItem {
    char*      name;
    double     price;
    int        qty;
    Attribute* attrs;   // linked list of attributes
    CartItem*  next;

    CartItem(const char* n, double p, int q)
        : price(p), qty(q), attrs(nullptr), next(nullptr) {
        name = new char[strlen(n) + 1];
        strcpy(name, n);
    }

    void addAttribute(const char* n, const char* v) {
        Attribute* a = new Attribute(n, v);
        a->next = attrs;
        attrs = a;   // prepend for O(1) insertion
    }
};

class ShoppingCart {
    CartItem* head;
    int       itemCount;
public:
    ShoppingCart() : head(nullptr), itemCount(0) {}
    ~ShoppingCart() { clear(); }

    void addItem(const char* name, double price, int qty);
    bool removeItem(const char* name);
    double calculateTotal() const;
    void sortByPrice();       // bubble sort on linked list
    void display() const;
    void clear();
};
// ThesaurusBasedWritingAssistant.cpp — Jagged Arrays + String Parsing

// Count synonym groups in the thesaurus data string
int countGroups(const char* data) {
    int count = 0;
    for (int i = 0; data[i] != '\0'; i++)
        if (data[i] == '|') count++;  // '|' separates groups
    return count + 1;
}

// Count words within a single group string
int countWords(const char* group) {
    int count = 0;
    bool inWord = false;
    for (int i = 0; group[i]; i++) {
        if (group[i] != ',' && !inWord) { count++; inWord = true; }
        else if (group[i] == ',')         inWord = false;
    }
    return count;
}

// Build 2D jagged array: thesaurus[group][word]
char*** CreateDynamicArray(const char* raw, int& gCount) {

// InventoryCleanup.cpp — Sparse Matrix → Compact List
// Concept: only store (row, col, value) for non-zero elements

struct CompactEntry {
    int row, col, value;
};

// Count how many non-zero elements exist in the grid
int countNonZero(int** grid, int rows, int cols) {
    int count = 0;
    for (int r = 0; r < rows; r++)
        for (int c = 0; c < cols; c++)
            if (grid[r][c] != 0) count++;
    return count;
}

// Build compact representation — drop all zeros
CompactEntry* createCompactList(int** grid, int rows, int cols, int& size) {
    size = countNonZero(grid, rows, cols);
    CompactEntry* list = new CompactEntry[size];
    int idx = 0;

    for (int r = 0; r < rows; r++)
        for (int c = 0; c < cols; c++)
            if (grid[r][c] != 0)
                list[idx++] = {r, c, grid[r][c]};

    return list;  // caller owns memory
}

// Print side-by-side memory comparison
void analyzeEfficiency(int rows, int cols, int nonZero) {
    int original = rows * cols * sizeof(int);
    int compact  = nonZero * sizeof(CompactEntry);
    double saved = 100.0 * (original - compact) / original;

    cout << "Original: "  << original << " bytes\n"
         << "Compact:  "  << compact  << " bytes\n"
         << "Saved:    "  << saved    << "%\n";
}

// 04 — Architecture

System
Design

The ecosystem uses a modular header-based architecture. Each sub-system is entirely self-contained in its own .cpp/.h pair, with the main hub acting as the single coordinator.

ECOSYSTEM
HUB
main.cpp
🍽️ Menu SystemMenuManagementSystem
🛒 Shopping CartShoppingCartMgmtSys
📖 ThesaurusThesaurusWritingAsst
📦 InventoryInventoryCleanup
🌍 EcosystemEcosystemSimulator
01
Header Separation

Every system exposes a clean .h interface. The hub includes only headers, keeping compilation boundaries sharp and modules reusable independently.

02
Single Entry Point

main.cpp serves as the unified launch pad with a loop-based numeric menu — one binary, five experiences. Clean dispatch via switch-case routing.

03
Static Linking

Compiled with -static flag — the final .exe contains all runtime dependencies, enabling zero-install deployment on any Windows machine.

What You
Learn

The ecosystem is an educational powerhouse — every sub-system is engineered to teach and reinforce a core pillar of C++ programming.

OOP
Object-Oriented Design

Classes with encapsulation, constructors/destructors, method overriding, and inheritance demonstrated throughout ShoppingCart and Ecosystem Simulator.

DS
Data Structures

Arrays (static, dynamic, jagged 2D), singly linked lists, sparse matrix compact lists, and STL vectors — each chosen to match the problem at hand.

MEM
Memory Management

Manual allocation/deallocation with new/delete[], ownership semantics, leak prevention through destructors, and memory efficiency analysis.

ALG
Algorithms

Bubble sort adapted for linked lists, multi-key comparison sort, linear search, binary search, string tokenization, and matrix compaction.

I/O
File I/O

Reading structured data from external .txt files (Q1.txt for menu data), parsing CSV-like formats, and robust input validation with error handling.

SYS
System Integration

Inter-system communication, unified user interface design, modular architecture with header files, and console color/animation using ANSI escape codes.
